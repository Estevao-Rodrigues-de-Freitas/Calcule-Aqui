/* script.js */
function calculateMode() {
    let numbersInput = document.getElementById("numbers").value;
    let separator = document.getElementById("separator").value || ",";
    let numbersArray = numbersInput.split(separator).map(num => parseFloat(num.trim())).filter(num => !isNaN(num));
    
    let frequency = {};
    let maxFreq = 0;
    let modes = [];
    
    numbersArray.forEach(num => {
        frequency[num] = (frequency[num] || 0) + 1;
        if (frequency[num] > maxFreq) {
            maxFreq = frequency[num];
            modes = [num];
        } else if (frequency[num] === maxFreq) {
            if (!modes.includes(num)) modes.push(num);
        }
    });
    
    document.getElementById("result").innerText = modes.join(", ");
}

function copyResult() {
    let resultText = document.getElementById("result").innerText;
    navigator.clipboard.writeText(resultText).then(() => {
        alert("Resultado copiado!");
    }).catch(err => {
        alert("Erro ao copiar: " + err);
    });
}
